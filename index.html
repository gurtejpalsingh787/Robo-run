using System.Collections;

public class GameManager : NetworkBehaviour
{
    [Header("Game Settings")]
    public float matchDuration = 300f; // 5 minutes
    public int maxPlayersPerTeam = 4;
    public Transform[] spawnPoints;
    
    [Header("Game State")]
    private NetworkVariable<GameState> currentState = 
        new NetworkVariable<GameState>(GameState.Lobby);
    
    private NetworkVariable<float> matchTimer = 
        new NetworkVariable<float>(0f);
    
    private Dictionary<ulong, PlayerData> connectedPlayers = 
        new Dictionary<ulong, PlayerData>();
    
    public static GameManager Instance { get; private set; }
    
    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(this);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);
    }
    
    public override void OnNetworkSpawn()
    {
        if (IsServer)
        {
            NetworkManager.Singleton.OnClientConnectedCallback += OnClientConnected;
            NetworkManager.Singleton.OnClientDisconnectCallback += OnClientDisconnected;
        }
        
        currentState.OnValueChanged += OnGameStateChanged;
        matchTimer.OnValueChanged += OnMatchTimerChanged;
    }
    
    private void OnClientConnected(ulong clientId)
    {
        Debug.Log($"Player connected: {clientId}");
        
        // Assign spawn point
        Vector3 spawnPos = GetAvailableSpawnPoint();
        SpawnPlayer(clientId, spawnPos);
        
        // Check if match can start
        if (connectedPlayers.Count >= 2 && currentState.Value == GameState.Lobby)
        {
            StartMatchCountdown();
        }
    }
    
    private void SpawnPlayer(ulong clientId, Vector3 position)
    {
        GameObject playerPrefab = NetworkManager.Singleton.NetworkConfig.PlayerPrefab;
        GameObject playerInstance = Instantiate(playerPrefab, position, Quaternion.identity);
        
        NetworkObject networkObject = playerInstance.GetComponent<NetworkObject>();
        networkObject.SpawnAsPlayerObject(clientId, true);
        
        // Initialize player data
        PlayerData playerData = new PlayerData
        {
            clientId = clientId,
            robotController = playerInstance.GetComponent<RobotController>(),
            score = 0,
            kills = 0,
            deaths = 0
        };
        
        connectedPlayers[clientId] = playerData;
    }
}

using UnityEngine;
using Unity.Netcode;
using System.Collections;

public class RobotController : NetworkBehaviour, IDamageable
{
    [Header("Robot Stats")]
    public RobotData robotData;
    
    [Header("Movement")]
    public float moveSpeed = 5f;
    public float jumpForce = 300f;
    public float dashDistance = 3f;
    
    [Header("Combat")]
    public Transform weaponSocket;
    public LayerMask enemyLayers;
    
    // Network Variables
    private NetworkVariable<float> networkHealth = new NetworkVariable<float>();
    private NetworkVariable<Vector3> networkPosition = new NetworkVariable<Vector3>();
    private NetworkVariable<Quaternion> networkRotation = new NetworkVariable<Quaternion>();
    
    // Components
    private Rigidbody rb;
    private Animator animator;
    private MobileInputManager inputManager;
    private AbilitySystem abilitySystem;
    
    // Input
    private Vector2 movementInput;
    private bool jumpInput;
    private bool dashInput;
    private bool isGrounded;
    
    // Dash system
    private float dashCooldown = 3f;
    private float lastDashTime;
    
    private void Awake()
    {
        rb = GetComponent<Rigidbody>();
        animator = GetComponentInChildren<Animator>();
        abilitySystem = GetComponent<AbilitySystem>();
    }
    
    public override void OnNetworkSpawn()
    {
        if (IsOwner)
        {
            inputManager = MobileInputManager.Instance;
            Camera.main.GetComponent<CameraFollow>().SetTarget(transform);
        }
        
        // Initialize health
        if (IsServer)
        {
            networkHealth.Value = robotData.maxHealth;
        }
        
        // Subscribe to network variable changes
        networkHealth.OnValueChanged += OnHealthChanged;
        networkPosition.OnValueChanged += OnPositionChanged;
        networkRotation.OnValueChanged += OnRotationChanged;
    }
    
    private void Update()
    {
        if (!IsOwner) return;
        
        HandleInput();
        HandleMovement();
        HandleCombat();
        
        // Send position updates to server
        UpdateServerPositionServerRpc(transform.position, transform.rotation);
    }
    
    private void HandleInput()
    {
        movementInput = inputManager.GetMovementInput();
        jumpInput = inputManager.GetJumpInput();
        dashInput = inputManager.GetDashInput();
    }
    
    private void HandleMovement()
    {
        // Ground check
        isGrounded = Physics.Raycast(transform.position, Vector3.down, 1.1f);
        
        // Movement
        Vector3 movement = new Vector3(movementInput.x, 0, movementInput.y);
        movement = Camera.main.transform.TransformDirection(movement);
        movement.y = 0;
        
        rb.MovePosition(transform.position + movement * moveSpeed * Time.deltaTime);
        
        // Rotation
        if (movement.magnitude > 0.1f)
        {
            Quaternion targetRotation = Quaternion.LookRotation(movement);
            transform.rotation = Quaternion.Slerp(transform.rotation, targetRotation, 10f * Time.deltaTime);
        }
        
        // Jump
        if (jumpInput && isGrounded)
        {
            rb.AddForce(Vector3.up * jumpForce);
            PlayAnimationServerRpc("Jump");
        }
        
        // Dash
        if (dashInput && Time.time - lastDashTime > dashCooldown)
        {
            PerformDash();
        }
        
        // Animation
        animator.SetFloat("Speed", movement.magnitude);
        animator.SetBool("IsGrounded", isGrounded);
    }
    
    private void PerformDash()
    {
        lastDashTime = Time.time;
        Vector3 dashDirection = transform.forward;
        
        rb.AddForce(dashDirection * dashDistance * 100f, ForceMode.Impulse);
        
        // Dash effects
        PlayDashEffectsServerRpc();
        
        // Brief invulnerability during dash
        StartCoroutine(DashInvulnerability());
    }
    
    [ServerRpc]
    private void UpdateServerPositionServerRpc(Vector3 position, Quaternion rotation)
    {
        networkPosition.Value = position;
        networkRotation.Value = rotation;
    }
    
    public void TakeDamage(float damage, ulong attackerId)
    {
        if (!IsServer) return;
        
        networkHealth.Value -= damage;
        
        if (networkHealth.Value <= 0)
        {
            Die(attackerId);
        }
    }
    
    private void Die(ulong killerId)
    {
        // Death logic
        PlayDeathEffectsClientRpc();
        
        // Respawn after delay
        StartCoroutine(RespawnCoroutine());
        
        // Update scores
        GameManager.Instance.HandlePlayerDeath(OwnerClientId, killerId);
    }
}

using System.Threading.Tasks;
using Unity.Netcode;
using Unity.Services.Authentication;
using Unity.Services.Lobbies;
using Unity.Services.Relay;

public class NetworkMatchmaker : MonoBehaviour
{
    [Header("Matchmaking Settings")]
    public int maxPlayersPerMatch = 8;
    public string gameMode = "TeamDeathMatch";
    
    private Lobby currentLobby;
    private string relayJoinCode;
    
    public static NetworkMatchmaker Instance { get; private set; }
    
    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(this);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);
    }
    
    public async Task InitializeServices()
    {
        try
        {
            await UnityServices.InitializeAsync();
            await AuthenticationService.Instance.SignInAnonymouslyAsync();
            
            Debug.Log("Unity Services initialized successfully");
        }
        catch (System.Exception e)
        {
            Debug.LogError($"Failed to initialize services: {e.Message}");
        }
    }
    
    public async Task StartMatchmaking()
    {
        try
        {
            // Try to find existing lobby
            List<Lobby> availableLobbies = await FindAvailableLobbies();
            
            if (availableLobbies.Count > 0)
            {
                // Join existing lobby
                await JoinLobby(availableLobbies[0].Id);
            }
            else
            {
                // Create new lobby
                await CreateLobby();
            }
        }
        catch (System.Exception e)
        {
            Debug.LogError($"Matchmaking failed: {e.Message}");
        }
    }
    
    private async Task<List<Lobby>> FindAvailableLobbies()
    {
        QueryLobbiesOptions options = new QueryLobbiesOptions
        {
            Filters = new List<QueryFilter>
            {
                new QueryFilter(QueryFilter.FieldOptions.AvailableSlots, "1", QueryFilter.OpOptions.GE),
                new QueryFilter(QueryFilter.FieldOptions.S1, gameMode, QueryFilter.OpOptions.EQ)
            }
        };
        
        QueryResponse response = await LobbyService.Instance.QueryLobbiesAsync(options);
        return response.Results;
    }
    
    private async Task CreateLobby()
    {
        // Create relay allocation
        Allocation allocation = await RelayService.Instance.CreateAllocationAsync(maxPlayersPerMatch - 1);
        relayJoinCode = await RelayService.Instance.GetJoinCodeAsync(allocation.AllocationId);
        
        // Create lobby
        CreateLobbyOptions options = new CreateLobbyOptions
        {
            IsPrivate = false,
            Data = new Dictionary<string, DataObject>
            {
                { "GameMode", new DataObject(DataObject.VisibilityOptions.Public, gameMode) },
                { "RelayCode", new DataObject(DataObject.VisibilityOptions.Member, relayJoinCode) }
            }
        };
        
        currentLobby = await LobbyService.Instance.CreateLobbyAsync("RoboRun Match", maxPlayersPerMatch, options);
        
        // Start as host
        StartHost(allocation);
        
        Debug.Log($"Created lobby: {currentLobby.Id}");
    }
    
    private async Task JoinLobby(string lobbyId)
    {
        currentLobby = await LobbyService.Instance.JoinLobbyByIdAsync(lobbyId);
        relayJoinCode = currentLobby.Data["RelayCode"].Value;
        
        // Join as client
        JoinAllocation joinAllocation = await RelayService.Instance.JoinAllocationAsync(relayJoinCode);
        StartClient(joinAllocation);
        
        Debug.Log($"Joined lobby: {currentLobby.Id}");
    }
    
    private void StartHost(Allocation allocation)
    {
        NetworkManager.Singleton.GetComponent<UnityTransport>().SetRelayServerData(allocation);
        NetworkManager.Singleton.StartHost();
    }
    
    private void StartClient(JoinAllocation joinAllocation)
    {
        NetworkManager.Singleton.GetComponent<UnityTransport>().SetRelayServerData(joinAllocation);
        NetworkManager.Singleton.StartClient();
    }
}

using UnityEngine;
using UnityEngine.EventSystems;

public class MobileInputManager : MonoBehaviour
{
    [Header("Virtual Joystick")]
    public VirtualJoystick movementJoystick;
    public VirtualJoystick aimJoystick;
    
    [Header("Action Buttons")]
    public VirtualButton jumpButton;
    public VirtualButton dashButton;
    public VirtualButton shootButton;
    public VirtualButton[] abilityButtons;
    
    [Header("Touch Settings")]
    public float swipeThreshold = 50f;
    public float tapTime = 0.2f;
    
    private Vector2 touchStartPos;
    private float touchStartTime;
    
    public static MobileInputManager Instance { get; private set; }
    
    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(gameObject);
            return;
        }
        Instance = this;
    }
    
    private void Start()
    {
        // Auto-detect platform and enable appropriate controls
        bool isMobile = Application.isMobilePlatform;
        SetMobileControlsActive(isMobile);
    }
    
    private void Update()
    {
        HandleTouchInput();
    }
    
    private void HandleTouchInput()
    {
        if (Input.touchCount > 0)
        {
            Touch touch = Input.GetTouch(0);
            
            switch (touch.phase)
            {
                case TouchPhase.Began:
                    OnTouchStart(touch.position);
                    break;
                    
                case TouchPhase.Ended:
                    OnTouchEnd(touch.position);
                    break;
                    
                case TouchPhase.Moved:
                    OnTouchMoved(touch.position);
                    break;
            }
        }
    }
    
    private void OnTouchStart(Vector2 position)
    {
        touchStartPos = position;
        touchStartTime = Time.time;
    }
    
    private void OnTouchEnd(Vector2 position)
    {
        float touchDuration = Time.time - touchStartTime;
        float touchDistance = Vector2.Distance(touchStartPos, position);
        
        if (touchDuration <= tapTime && touchDistance < swipeThreshold)
        {
            // Handle tap
            HandleTap(position);
        }
        else if (touchDistance >= swipeThreshold)
        {
            // Handle swipe
            Vector2 swipeDirection = (position - touchStartPos).normalized;
            HandleSwipe(swipeDirection);
        }
    }
    
    private void HandleTap(Vector2 position)
    {
        // Check if tap is on UI element
        if (EventSystem.current.IsPointerOverGameObject())
            return;
        
        // Auto-aim or shoot at tap position
        HandleAutoAim(position);
    }
    
    private void HandleSwipe(Vector2 direction)
    {
        // Swipe gestures for special abilities
        if (Mathf.Abs(direction.y) > Mathf.Abs(direction.x))
        {
            if (direction.y > 0)
            {
                // Swipe up - Jump ability
                TriggerAbility(0);
            }
            else
            {
                // Swipe down - Ground slam
                TriggerAbility(1);
            }
        }
        else
        {
            if (direction.x > 0)
            {
                // Swipe right - Dash right
                TriggerDash(Vector3.right);
            }
            else
            {
                // Swipe left - Dash left
                TriggerDash(Vector3.left);
            }
        }
    }
    
    public Vector2 GetMovementInput()
    {
        return movementJoystick != null ? movementJoystick.Direction : Vector2.zero;
    }
    
    public Vector2 GetAimInput()
    {
        return aimJoystick != null ? aimJoystick.Direction : Vector2.zero;
    }
    
    public bool GetJumpInput()
    {
        return jumpButton != null ? jumpButton.IsPressed : Input.GetKeyDown(KeyCode.Space);
    }
    
    public bool GetDashInput()
    {
        return dashButton != null ? dashButton.IsPressed : Input.GetKeyDown(KeyCode.LeftShift);
    }
}

[System.Serializable]
public class RobotData
{
    public string robotName;
    public int level = 1;
    public int experience = 0;
    public float maxHealth = 100f;
    public float attackPower = 25f;
    public float speed = 5f;
    public float defense = 10f;

    public CustomizationData customization = new CustomizationData();
}

public class ProgressionSystem : MonoBehaviour
{
    [Header("Progression Settings")]
    public AnimationCurve experienceRequiredCurve;
    public float baseExperienceRequired = 100f;
    public int maxLevel = 100;
    
    [Header("Stat Growth")]
    public float healthGrowthPerLevel = 10f;
    public float attackGrowthPerLevel = 2f;
    public float defenseGrowthPerLevel = 1f;
    
    [Header("Unlock Levels")]
    public LevelUnlock[] levelUnlocks;
    
    private Dictionary<string, RobotData> playerRobots = new Dictionary<string, RobotData>();
    
    public static ProgressionSystem Instance { get; private set; }
    
    public event System.Action<RobotData, int> OnLevelUp;
    public event System.Action<RobotData, int> OnExperienceGained;
    
    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(this);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);
        
        LoadPlayerData();
    }
    
    public void AddExperience(string robotId, int experience)
    {
        if (!playerRobots.ContainsKey(robotId))
        {
            Debug.LogWarning($"Robot {robotId} not found!");
            return;
        }
        
        RobotData robot = playerRobots[robotId];
        int oldLevel = robot.level;
        
        robot.experience += experience;
        OnExperienceGained?.Invoke(robot, experience);
        
        // Check for level up
        int experienceRequired = GetExperienceRequiredForLevel(robot.level + 1);
        
        while (robot.experience >= experienceRequired && robot.level < maxLevel)
        {
            LevelUp(robot);
            experienceRequired = GetExperienceRequiredForLevel(robot.level + 1);
        }
        
        SavePlayerData();
    }
    
    private void LevelUp(RobotData robot)
    {
        int experienceRequired = GetExperienceRequiredForLevel(robot.level);
        robot.experience -= experienceRequired;
        robot.level++;
        
        // Increase stats
        robot.maxHealth += healthGrowthPerLevel;
        robot.attackPower += attackGrowthPerLevel;
        robot.defense += defenseGrowthPerLevel;
        
        // Check for unlocks
        CheckLevelUnlocks(robot);
        
        OnLevelUp?.Invoke(robot, robot.level);
        
        Debug.Log($"{robot.robotName} leveled up to {robot.level}!");
    }
    
    private void CheckLevelUnlocks(RobotData robot)
    {
        foreach (LevelUnlock unlock in levelUnlocks)
        {
            if (unlock.level == robot.level)
            {
                switch (unlock.unlockType)
                {
                    case UnlockType.Ability:
                        UnlockAbility(robot, unlock.abilityData);
                        break;
                        
                    case UnlockType.Customization:
                        UnlockCustomization(robot, unlock.customizationItem);
                        break;
                        
                    case UnlockType.GameMode:
                        UnlockGameMode(unlock.gameMode);
                        break;
                }
            }
        }
    }
    
    public int GetExperienceRequiredForLevel(int level)
    {
        if (level <= 1) return 0;
        
        float curveValue = experienceRequiredCurve.Evaluate((float)level / maxLevel);
        return Mathf.RoundToInt(baseExperienceRequired * curveValue * level);
    }
    
    public float GetProgressToNextLevel(RobotData robot)
    {
        if (robot.level >= maxLevel) return 1f;
        
        int experienceRequired = GetExperienceRequiredForLevel(robot.level + 1);
        return (float)robot.experience / experienceRequired;
    }
}

[System.Serializable]
public class BattlePassTier
{
    public int tier;
    public int requiredXP;
    public RewardData freeReward;
    public RewardData premiumReward;
}

public class BattlePassSystem : MonoBehaviour
{
    [Header("Battle Pass Settings")]
    public int maxTier = 100;
    public float seasonDuration = 90f; // days
    public BattlePassTier[] tiers;
    
    [Header("XP Sources")]
    public int xpPerMatch = 50;
    public int xpPerKill = 10;
    public int xpPerWin = 100;
    public int dailyChallengeXP = 200;
    
    private BattlePassData currentSeasonData;
    
    public static BattlePassSystem Instance { get; private set; }
    
    public event System.Action<int, RewardData> OnRewardUnlocked;
    public event System.Action<int> OnTierReached;
    public event System.Action<int> OnXPGained;
    
    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(this);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);
    }
    
    private void Start()
    {
        LoadBattlePassData();
        CheckSeasonExpiry();
    }
    
    public void AddBattlePassXP(int xp, string source)
    {
        if (currentSeasonData == null) return;
        
        int oldTier = GetCurrentTier();
        currentSeasonData.currentXP += xp;
        int newTier = GetCurrentTier();
        
        OnXPGained?.Invoke(xp);
        
        // Check for tier progression
        for (int tier = oldTier + 1; tier <= newTier; tier++)
        {
            UnlockTierRewards(tier);
        }
        
        SaveBattlePassData();
        
        Debug.Log($"Gained {xp} XP from {source}. Total: {currentSeasonData.currentXP}");
    }
    
    private void UnlockTierRewards(int tier)
    {
        if (tier > maxTier || tier < 1) return;
        
        BattlePassTier tierData = tiers[tier - 1];
        
        // Free reward
        if (tierData.freeReward != null)
        {
            GiveReward(tierData.freeReward);
            OnRewardUnlocked?.Invoke(tier, tierData.freeReward);
        }
        
        // Premium reward (if purchased)
        if (currentSeasonData.isPremium && tierData.premiumReward != null)
        {
            GiveReward(tierData.premiumReward);
            OnRewardUnlocked?.Invoke(tier, tierData.premiumReward);
        }
        
        OnTierReached?.Invoke(tier);
    }
    
    private void GiveReward(RewardData reward)
    {
        switch (reward.rewardType)
        {
            case RewardType.Currency:
                CurrencyManager.Instance.AddCurrency(reward.currencyType, reward.amount);
                break;
                
            case RewardType.Cosmetic:
                CosmeticManager.Instance.UnlockCosmetic(reward.cosmeticId);
                break;
                
            case RewardType.XPBoost:
                BuffManager.Instance.ApplyXPBoost(reward.duration, reward.multiplier);
                break;
                
            case RewardType.RobotPart:
                InventoryManager.Instance.AddItem(reward.itemId, reward.amount);
                break;
        }
    }
    
    public int GetCurrentTier()
    {
        if (currentSeasonData == null) return 0;
        
        int tier = 0;
        int totalXP = 0;
        
        for (int i = 0; i < tiers.Length; i++)
        {
            totalXP += tiers[i].requiredXP;
            if (currentSeasonData.currentXP >= totalXP)
            {
                tier = i + 1;
            }
            else
            {
                break;
            }
        }
        
        return Mathf.Min(tier, maxTier);
    }
    
    public float GetProgressToNextTier()
    {
        int currentTier = GetCurrentTier();
        if (currentTier >= maxTier) return 1f;
        
        int xpForCurrentTier = GetTotalXPForTier(currentTier);
        int xpForNextTier = GetTotalXPForTier(currentTier + 1);
        
        int progressXP = currentSeasonData.currentXP - xpForCurrentTier;
        int requiredXP = xpForNextTier - xpForCurrentTier;
        
        return (float)progressXP / requiredXP;
    }
    
    public void PurchasePremiumBattlePass()
    {
        if (currentSeasonData == null) return;
        
        currentSeasonData.isPremium = true;
        
        // Retroactively give premium rewards for already unlocked tiers
        int currentTier = GetCurrentTier();
        for (int i = 1; i <= currentTier; i++)
        {
            BattlePassTier tierData = tiers[i - 1];
            if (tierData.premiumReward != null)
            {
                GiveReward(tierData.premiumReward);
            }
        }
        
        SaveBattlePassData();
        Debug.Log("Premium Battle Pass purchased!");
    }
}

using UnityEngine;

    private void DetectDeviceCapabilities()
    {
        int memoryMB = SystemInfo.systemMemorySize;
        int vramMB = SystemInfo.graphicsMemorySize;
        string gpu = SystemInfo.graphicsDeviceName;
        
        // Auto-select quality based on device specs
        if (memoryMB < 3000 || vramMB < 1000)
        {
            currentQualityIndex = 0; // Low
        }
        else if (memoryMB < 6000 || vramMB < 2000)
        {
            currentQualityIndex = 1; // Medium
        }
        else
        {
            currentQualityIndex = 2; // High
        }
        
        ApplyQualityProfile(currentQualityIndex);
        
        Debug.Log($"Device: {memoryMB}MB RAM, {vramMB}MB VRAM, GPU: {gpu}");
        Debug.Log($"Selected Quality: {qualityProfiles[currentQualityIndex].profileName}");
    }
    
    private void ApplyQualityProfile(int profileIndex)
    {
        if (profileIndex < 0 || profileIndex >= qualityProfiles.Length) return;
        
        QualityProfile profile = qualityProfiles[profileIndex];
        
        // Apply settings
        QualitySettings.SetQualityLevel(profile.unityQualityLevel);
        
        // Custom settings
        QualitySettings.shadowResolution = profile.shadowResolution;
        QualitySettings.shadowDistance = profile.shadowDistance;
        QualitySettings.particleRaycastBudget = profile.particleBudget;
        
        // Texture settings
        QualitySettings.masterTextureLimit = profile.textureQuality;
        QualitySettings.anisotropicFiltering = profile.anisotropicFiltering;
    }
    
    private IEnumerator MonitorPerformance()
    {
        while (true)
        {
            UpdatePerformanceMetrics();
            
            if (enableAutoOptimization)
            {
                CheckPerformanceThresholds();
            }
            
            yield return new WaitForSeconds(monitoringInterval);
        }
    }
    
    private void UpdatePerformanceMetrics()
    {
        frameTime += (Time.unscaledDeltaTime - frameTime) * 0.1f;
        currentFPS = 1f / frameTime;
        memoryUsage = (System.GC.GetTotalMemory(false) / 1024f / 1024f / 1024f);
    }
    
    private void CheckPerformanceThresholds()
    {
        // FPS below threshold - reduce quality
        if (currentFPS < fpsThreshold && currentQualityIndex > 0)
        {
            currentQualityIndex--;
            ApplyQualityProfile(currentQualityIndex);
            Debug.Log($"Performance drop detected. Quality reduced to: {qualityProfiles[currentQualityIndex].profileName}");
        }
        
        // Memory usage high - force garbage collection
        if (memoryUsage > memoryThreshold)
        {
            ForceGarbageCollection();
        }
    }
    
    private void InitializeObjectPools()
    {
        bulletPool = new ObjectPool("BulletPool", 100);
        effectPool = new ObjectPool("EffectPool", 50);
        robotPool = new ObjectPool("RobotPool", 20);
    }
    
    public void ForceGarbageCollection()
    {
        System.GC.Collect();
        Resources.UnloadUnusedAssets();
        Debug.Log("Forced garbage collection completed");
    }
}

using System.Threading.Tasks;
using UnityEngine;
using System.Text;
using UnityEngine.Networking;

public class DatabaseManager : MonoBehaviour
{
    [Header("Server Configuration")]
    public string serverURL = "https://api.roborun.com";
    public string apiKey;
    public float timeout = 10f;
    
    private string playerToken;
    private PlayerProfile currentPlayer;
    
    public static DatabaseManager Instance { get; private set; }
    
    private void Awake()
    {
        if (Instance != null && Instance != this)
        {
            Destroy(this);
            return;
        }
        Instance = this;
        DontDestroyOnLoad(gameObject);
    }
    
    public async Task<bool> AuthenticatePlayer(string deviceId)
    {
        string url = $"{serverURL}/auth/login";
        
        AuthRequest authData = new AuthRequest
        {
            deviceId = deviceId,
            platform = Application.platform.ToString(),
            version = Application.version
        };
        
        string jsonData = JsonUtility.ToJson(authData);
        byte[] bodyRaw = Encoding.UTF8.GetBytes(jsonData);
        
        using (UnityWebRequest request = new UnityWebRequest(url, "POST"))
        {
            request.uploadHandler = new UploadHandlerRaw(bodyRaw);
            request.downloadHandler = new DownloadHandlerBuffer();
            request.SetRequestHeader("Content-Type", "application/json");
            request.SetRequestHeader("X-API-Key", apiKey);
            request.timeout = (int)timeout;
            
            var operation = request.SendWebRequest();
            
            while (!operation.isDone)
                await Task.Yield();
            
            if (request.result == UnityWebRequest.Result.Success)
            {
                AuthResponse response = JsonUtility.FromJson<AuthResponse>(request.downloadHandler.text);
                playerToken = response.token;
                currentPlayer = response.player;
                
                Debug.Log("Player authenticated successfully");
                return true;
            }
            else
            {
                Debug.LogError($"Authentication failed: {request.error}");
                return false;
            }
        }
    }
    
    public async Task<bool> SavePlayerData(PlayerProfile player)
    {
        if (string.IsNullOrEmpty(playerToken))
        {
            Debug.LogError("No player token. Please authenticate first.");
            return false;
        }
        
        string url = $"{serverURL}/player/save";
        string jsonData = JsonUtility.ToJson(player);
        byte[] bodyRaw = Encoding.UTF8.GetBytes(jsonData);
        
        using (UnityWebRequest request = new UnityWebRequest(url, "POST"))
        {
            request.uploadHandler = new UploadHandlerRaw(bodyRaw);
            request.downloadHandler = new DownloadHandlerBuffer();
            request.SetRequestHeader("Content-Type", "application/json");
            request.SetRequestHeader("Authorization", $"Bearer {playerToken}");
            
            var operation = request.SendWebRequest();
            
            while (!operation.isDone)
                await Task.Yield();
            
            return request.result == UnityWebRequest.Result.Success;
        }
    }
    
    public async Task<LeaderboardData> GetLeaderboard(LeaderboardType type, int limit = 100)
    {
        string url = $"{serverURL}/leaderboard/{type.ToString().ToLower()}?limit={limit}";
        
        using (UnityWebRequest request = UnityWebRequest.Get(url))
        {
            request.SetRequestHeader("Authorization", $"Bearer {playerToken}");
            
            var operation = request.SendWebRequest();
            
            while (!operation.isDone)
                await Task.Yield();
            
            if (request.result == UnityWebRequest.Result.Success)
            {
                return JsonUtility.FromJson<LeaderboardData>(request.downloadHandler.text);
            }
            
            return null;
        }
    }
    
    public async Task<bool> SubmitMatchResult(MatchResult result)

        }
    }
